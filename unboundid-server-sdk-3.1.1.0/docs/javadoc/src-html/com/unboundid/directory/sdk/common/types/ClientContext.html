027    package com.unboundid.directory.sdk.common.types;
028    
029    
030    
031    import java.net.InetAddress;
032    
033    import com.unboundid.ldap.sdk.ResultCode;
034    import com.unboundid.util.NotExtensible;
035    import com.unboundid.util.ThreadSafety;
036    import com.unboundid.util.ThreadSafetyLevel;
037    
038    
039    
040    /**
041     * This interface defines a set of methods that may be used to obtain
042     * information about a client connection that has been established to the
043     * server.
044     */
045    @NotExtensible()
046    @ThreadSafety(level=ThreadSafetyLevel.INTERFACE_NOT_THREADSAFE)
047    public interface ClientContext
048    {
049      /**
050       * Retrieves the identifier that has been assigned to the associated client
051       * connection.
052       *
053       * @return  The identifier that has been assigned to the associated client
054       *          connection.
055       */
056      long getConnectionID();
057    
058    
059    
060      /**
061       * Indicates whether this represents an internal client connection.
062       *
063       * @return  {@code true} if this represents an internal client connection, or
064       *          {@code false} if it is from an external client.
065       */
066      boolean isInternal();
067    
068    
069    
070      /**
071       * Indicates whether the client is communicating with the server in a secure
072       * manner.
073       *
074       * @return  {@code true} if the client is communicating with the server in a
075       *          secure manner, or {@code false} if not.
076       */
077      boolean isSecure();
078    
079    
080    
081      /**
082       * Retrieves the name of the protocol that the client is using to communicate
083       * with the server.
084       *
085       * @return  The name of the protocol that the client is using to communicate
086       *          with the server.
087       */
088      String getProtocol();
089    
090    
091    
092      /**
093       * Retrieves the time that the connection was established.  The value returned
094       * will be an offset in milliseconds since 12:00 a.m. on January 1, 1970.
095       *
096       * @return  The time that the connection was established.
097       */
098      long getConnectTime();
099    
100    
101    
102      /**
103       * Retrieves an {@code InetAddress} representing the address of the client
104       * system, if available.
105       *
106       * @return  An {@code InetAddress} representing the address of the client
107       *          system, or {@code null} if that is not available or applicable for
108       *          the associated client connection.
109       */
110      InetAddress getClientInetAddress();
111    
112    
113    
114      /**
115       * Retrieves an {@code InetAddress} representing the address on the server to
116       * which the client established the connection, if available.
117       *
118       * @return  The address on the server to which the client established the
119       *          connection, or {@code null} if that is not available or
120       *          applicable.
121       */
122      InetAddress getServerInetAddress();
123    
124    
125    
126      /**
127       * Indicates whether the client has authenticated to the server.
128       *
129       * @return  {@code true} if the client has authenticated to the server, or
130       *          {@code false} if not.
131       */
132      boolean isAuthenticated();
133    
134    
135    
136      /**
137       * Retrieves information about the authentication state of the client
138       * connection.
139       *
140       * @return  Information about the authentication state of the client
141       *          connection.
142       */
143      AuthInfo getAuthInfo();
144    
145    
146    
147      /**
148       * Attempts to send an unsolicited notification to the client with the
149       * provided information.
150       *
151       * @param  oid         The OID for the unsolicited notification.  It must not
152       *                     be {@code null}.
153       * @param  resultCode  The result code to use for the unsolicited
154       *                     notification.  It must not be {@code null}.
155       * @param  message     A message to include in the unsolicited notification.
156       *                     It may be {@code null} if no message is needed.
157       */
158      void sendUnsolicitedNotification(final String oid,
159                                       final ResultCode resultCode,
160                                       final String message);
161    
162    
163    
164      /**
165       * Terminates the connection to the client and interrupts any operations that
166       * may be in progress on that connection.
167       *
168       * @param  reason        A general reason that the connection was closed.
169       * @param  notifyClient  Indicates whether to attempt to send a notice of
170       *                       disconnection to the client.
171       * @param  message       A message with information about the reason for the
172       *                       disconnect.  It may be {@code null} if none is
173       *                       available.  It is generally recommended that a
174       *                       message be provided even if the client should not be
175       *                       notified, since the message may be used in other
176       *                       ways (e.g., in log messages).
177       */
178      void disconnect(final DisconnectReason reason, final boolean notifyClient,
179                      final String message);
180    
181    
182    
183      /**
184       * Retrieves information about the server with which the client connection is
185       * associated.
186       *
187       * @return  Information about the server with which the client connection is
188       *          associated.
189       */
190      ServerContext getServerContext();
191    
192    
193    
194      /**
195       * Retrieves a string representation of the client connection.
196       *
197       * @return  A string representation of the client connection.
198       */
199      String toString();
200    }
