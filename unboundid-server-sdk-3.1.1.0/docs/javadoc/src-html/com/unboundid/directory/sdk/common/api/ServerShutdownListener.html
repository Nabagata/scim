027    package com.unboundid.directory.sdk.common.api;
028    
029    
030    
031    import com.unboundid.directory.sdk.ds.internal.DirectoryServerExtension;
032    import com.unboundid.directory.sdk.proxy.internal.DirectoryProxyServerExtension;
033    import com.unboundid.directory.sdk.sync.internal.SynchronizationServerExtension;
034    import com.unboundid.util.Extensible;
035    import com.unboundid.util.ThreadSafety;
036    import com.unboundid.util.ThreadSafetyLevel;
037    
038    
039    
040    /**
041     * This interface defines a set of methods that should be implemented by classes
042     * which should be notified when the server has begun the process of shutting
043     * down.  Shutdown listeners should be registered with the server using the
044     * {@code ServerContext.registerShutdownListener} method, and the corresponding
045     * {@code deregisterShutdownListener} should be used to deregister the listener
046     * if it is no longer needed.
047     * <BR><BR>
048     * Most types of extensions will not need to implement this interface, since
049     * they may use their own finalization method to be notified when the extension
050     * is to be taken out of service (either because the server is shutting down or
051     * because the extension is being disabled or removed).  However, it may be
052     * useful for background threads that may not be tied to any single instance of
053     * an extension.
054     */
055    @Extensible()
056    @DirectoryServerExtension()
057    @DirectoryProxyServerExtension(appliesToLocalContent=true,
058         appliesToRemoteContent=false)
059    @SynchronizationServerExtension(appliesToLocalContent=true,
060         appliesToSynchronizedContent=false)
061    @ThreadSafety(level=ThreadSafetyLevel.INTERFACE_NOT_THREADSAFE)
062    public interface ServerShutdownListener
063    {
064      /**
065       * Performs any processing that may be necessary when the server begins the
066       * shutdown process.
067       *
068       * @param  reason  Information about the reason for the shutdown.
069       */
070      void processServerShutdown(final String reason);
071    }
