027    package com.unboundid.directory.sdk.sync.types;
028    
029    import java.util.Collection;
030    import java.util.List;
031    
032    import com.unboundid.ldap.sdk.ChangeLogEntry;
033    import com.unboundid.ldap.sdk.Entry;
034    
035    /**
036     * This interface represents a single synchronized change from the Sync
037     * Source to the Sync Destination.  It exists from when a change is detected
038     * all the way through when the change is applied at the destination.
039     */
040    public interface SyncOperation
041    {
042      /**
043       * Return the type of this SynchronizationOperation.
044       *
045       * @return  The type of this SynchronizationOperation.
046       */
047      SyncOperationType getType();
048    
049    
050    
051      /**
052       * Return a String that can be used to correlate this change back to the
053       * source entry. For example, if the source endpoint is a directory server,
054       * this will be a DN.
055       *
056       * @return  A String that can be used to correlate this change back to the
057       *          source entry.
058       */
059      String getIdentifiableInfo();
060    
061    
062    
063      /**
064       * Return a unique ID for this {@code SyncOperation}. This ID is used in
065       * log messages.
066       *
067       * @return  A unique ID for this operation.
068       */
069      String getUniqueId();
070    
071    
072    
073      /**
074       * Return the Entry that was fetched from the sync source. This may be null
075       * if the source entry no longer exists, or if the Sync Pipe is in
076       * notification mode.
077       *
078       * @return  The entry entry that was fetched from the source server.
079       */
080      Entry getSourceEntry();
081    
082    
083    
084      /**
085       * Sets the entry that was fetched from the sync source.  If the entry was
086       * deleted, then sufficient information about the entry must be provided so
087       * that it can be correlated to a destination entry.
088       *
089       * @param  sourceEntry     The entry that was fetched from the source.
090       * @param  isDeletedEntry  {@code true} if and only if the entry was deleted
091       *                         at the source.
092       */
093      void setSourceEntry(Entry sourceEntry, boolean isDeletedEntry);
094    
095    
096    
097      /**
098       * Return the DatabaseChangeRecord that was detected from a RDBMS. This is
099       * only applicable when synchronizing from a database environment, and will
100       * be null otherwise.
101       *
102       * @return a DatabaseChangeRecord that was detected from a database, or null
103       *         if the sync source is not a JDBC sync source.
104       */
105      DatabaseChangeRecord getDatabaseChangeRecord();
106    
107    
108    
109      /**
110       * Return the original changelog entry that was detected for this operation.
111       * This is only applicable when synchronizing from a data source that
112       * supplies an LDAP changelog as described in
113       * <i>draft-good-ldap-changelog</i>, such as the UnboundID Directory Server.
114       *
115       * @return a {@link ChangeLogEntry} instance, or null if the source endpoint
116       *         does not use an LDAP changelog for change detection.
117       */
118      ChangeLogEntry getChangeLogEntry();
119    
120    
121    
122      /**
123       * Determines whether this SyncOperation represents a modify DN operation on
124       * the source entry (if the source is an LDAP directory). The
125       * result of this method is undefined if the source is a relational database
126       * or some other type of endpoint.
127       *
128       * @return true if the entry has been renamed, false otherwise.
129       */
130      boolean isModifyDN();
131    
132    
133    
134      /**
135       * Returns a representation of the target destination entry before it is
136       * modified or renamed. This can be used to obtain the before-values of any
137       * attributes. The returned entry cannot be modified.
138       * <p>
139       * This value is only defined after the SyncOperation has made it past the
140       * post-mapping stage in the SyncPipe.
141       *
142       * @return a {@link Entry} instance.
143       */
144      Entry getDestinationEntryBeforeChange();
145    
146    
147    
148      /**
149       * Returns a representation of the target destination entry after it is
150       * modified or renamed. This can be used to obtain the after-values of any
151       * attributes.
152       * <p>
153       * This value is only defined after the SyncOperation has made it past the
154       * post-mapping stage in the SyncPipe.
155       * <p>
156       * The returned entry is modifiable and can be used to influence
157       * the final list of modifications to apply at the destination. The
158       * Synchronization Server computes a diff between this entry and the existing
159       * entry returned from the destination. Any modifications to this entry would
160       * have to take place in either {@code LDAPSyncDestinationPlugin.preFetch} or
161       * {@code LDAPSyncDestinationPlugin.postFetch}.
162       *
163       * @return a {@link Entry} instance.
164       */
165      Entry getDestinationEntryAfterChange();
166    
167    
168    
169      /**
170       * Return the names of the attributes that were modified at the source.
171       *
172       * @return  The names of the attributes that were modified at the source.
173       */
174      Collection<String> getModifiedSourceAttributes();
175    
176    
177    
178      /**
179       * Add an additional source attribute that should be synchronized to the
180       * destination.
181       *
182       * @param  attributeName  The attribute to add to the set of attributes to
183       *                        synchronize.
184       */
185      void addModifiedSourceAttribute(String attributeName);
186    
187    
188    
189      /**
190       * Remove a source attribute that should not be synchronized to the
191       * destination.
192       *
193       * @param  attributeName  The attribute to remove from the set of attributes
194       *                        to synchronize.
195       */
196      void removeModifiedSourceAttribute(String attributeName);
197    
198    
199    
200      /**
201       * Add the name of a destination attribute that should be synchronized.
202       *
203       * @param  attributeName  The name of the destination attribute to
204       *                        synchronize.
205       */
206      void addModifiedDestinationAttribute(String attributeName);
207    
208    
209    
210      /**
211       * Return the name of the Sync Pipe for the operation being processed.
212       *
213       * @return  The name of the Sync Pipe for the operation being processed.
214       */
215      String getSyncPipeName();
216    
217    
218    
219      /**
220       * Return the name of the Sync Class for the operation being processed.
221       *
222       * @return  The name of the Sync Class for the operation being processed.
223       */
224      String getSyncClassName();
225    
226    
227    
228      /**
229       * Return the {@link SyncClass} for the operation being processed. This can
230       * be used to determine the correlation attributes and other properties
231       * configured for this type of change.
232       *
233       * @return  The Sync Class that this operation falls into, or null if no
234       *          Sync Class has been determined yet.
235       */
236      SyncClass getSyncClass();
237    
238    
239    
240      /**
241       * Stores an attachment in the operation.  This can be used by an extension
242       * to have information carried in the synchronization operation from one
243       * extension point to the next.
244       *
245       * @param  key    The key of the attachment.  To avoid conflicts, the
246       *                fully-qualified class name of the extension should be used
247       *                as a prefix of the key.
248       *
249       * @param  value  The value of the attachment.
250       */
251      void putAttachment(Object key, Object value);
252    
253    
254    
255      /**
256       * Returns an attachment from the operation that was previously set with a
257       * call to {@link #putAttachment}.
258       *
259       * @param  key  The key of the attachment.
260       *
261       * @return  The value of the attachment or {@code null} if no attachment was
262       *          set for the specified key.
263       */
264      Object getAttachment(Object key);
265    
266    
267    
268      /**
269       * Returns a list of warning messages (if any) about this SyncOperation. For
270       * example, if a changelog entry omitted some data because an attribute had
271       * too many values, there will be a message in the returned list indicating
272       * so. Typically synchronization of this SyncOperation can continue despite
273       * these warnings.
274       *
275       * @return a list of warning messages; the list will be empty if there are no
276       *         warnings.
277       */
278      List<String> getWarnings();
279    
280    
281    
282      /**
283       * Logs an error message to the synchronization log for this change.  Messages
284       * logged here should pertain to this specific change.  For general logging
285       * capabilities, see {@link SyncServerContext}
286       *
287       * @param  message  The message to log.
288       */
289      void logError(String message);
290    
291    
292    
293      /**
294       * Logs an informational message to the synchronization log for this change.
295       * Messages logged here should pertain to this specific change.  For general
296       * logging capabilities, see {@link SyncServerContext}
297       *
298       * @param  message  The message to log.
299       */
300      void logInfo(String message);
301    }
