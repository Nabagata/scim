046    /**
047     * This class defines an API that must be implemented by scripted extensions
048     * in order to synchronize data into a relational database. Since the UnboundID
049     * Synchronization Server is LDAP-centric,
050     * this API allows you to take LDAP entries and split them out into
051     * database content and make the appropriate updates. The lifecycle of a sync
052     * operation is as follows:
053     * <ol>
054     * <li>Detect change at the synchronization source</li>
055     * <li>Fetch full source entry</li>
056     * <li>Perform any mappings and compute the equivalent destination entry</li>
057     * <li>Fetch full destination entry</li>
058     * <li>Diff the computed destination entry and actual destination entry</li>
059     * <li>Apply the minimal set of changes at the destination to bring it in sync
060     * </li>
061     * </ol>
062     * This implies that the {@link #fetchEntry(TransactionContext, Entry,
063     * SyncOperation)} method will
064     * always be called once prior to any of the other methods in this class.
065     * <p>
066     * In several places a {@link TransactionContext} is provided, which allows
067     * controlled access to the target database. By default, methods in this class
068     * are always provided with a fresh connection (i.e. a new transaction), and the
069     * Synchronization Server will always commit or rollback the transaction
070     * automatically, depending on whether the method returned normally or threw an
071     * exception. Implementers may optionally perform their own transaction
072     * management within these methods if necessary. The TransactionContext also
073     * provides logging facilities and allows a timeout, which will automatically
074     * invalidate the context object after the specified duration.
075     * <p>
076     * Several of these methods throw {@link SQLException}, which should be used in
077     * the case of any database access error. For other types of errors, runtime
078     * exceptions may be used (IllegalStateException, NullPointerException, etc.).
079     * The Synchronization Server will automatically retry operations that fail, up
080     * to a configurable amount of attempts. The exception to this rule is if a
081     * SQLException is thrown with a SQL state string beginning with "08"; this
082     * indicates a connection error, and in this case the operation is retried
083     * indefinitely.
084     * <p>
085     * Implementers need not worry about the cost of repeatedly parsing and
086     * preparing SQL statements; all statements will be automatically cached.
087     * <BR>
088     * <H2>Configuring Groovy-Scripted JDBC Sync Destinations</H2>
089     * In order to configure a scripted JDBC sync destination based on this API and
090     * written in the Groovy scripting language, use a command like:
091     * <PRE>
092     *      dsconfig create-sync-destination \
093     *           --destination-name "<I>{destination-name}</I>" \
094     *           --type groovy-scripted-jdbc \
095     *           --set "server:{server-name}" \
096     *           --set "script-class:<I>{class-name}</I>" \
097     *           --set "script-argument:<I>{name=value}</I>"
098     * </PRE>
099     * where "<I>{destination-name}</I>" is the name to use for the JDBC sync
100     * destination instance, "<I>{server-name}</I>" is the name of the JDBC external
101     * server that will be used as the sync destination, "<I>{class-name}</I>" is
102     * the fully-qualified name of the Groovy class written using this API, and
103     * "<I>{name=value}</I>" represents name-value pairs for any arguments to
104     * provide to the JDBC sync destination.  If multiple arguments should be
105     * provided to the JDBC sync destination, then the
106     * "<CODE>--set script-argument:<I>{name=value}</I></CODE>" option should be
107     * provided multiple times.
108     */
109    @Extensible()
110    @SynchronizationServerExtension(appliesToLocalContent=false,
111                                    appliesToSynchronizedContent=true)
112    public abstract class ScriptedJDBCSyncDestination implements Configurable
113    {
114      /**
115       * {@inheritDoc}
116       */
117      public void defineConfigArguments(final ArgumentParser parser)
118             throws ArgumentException
119      {
120        // No arguments will be allowed by default.
121      }
122    
123      /**
124       * This hook is called when a Sync Pipe first starts up, or when the Resync
125       * process first starts up. Any initialization should be performed here.
126       * @param ctx
127       *          a TransactionContext which provides a valid JDBC connection to the
128       *          database.
129       * @param  serverContext  A handle to the server context for the server in
130       *                        which this extension is running.
131       * @param  config         The general configuration for this sync destination.
132       * @param  parser         The argument parser which has been initialized from
133       *                        the configuration for this JDBC sync destination.
134       */
135      @ThreadSafety(level = ThreadSafetyLevel.METHOD_NOT_THREADSAFE)
136      public void initializeJDBCSyncDestination(
137                                             final TransactionContext ctx,
138                                             final SyncServerContext serverContext,
139                                             final JDBCSyncDestinationConfig config,
140                                             final ArgumentParser parser)
141      {
142        // No initialization will be performed by default.
143      }
144    
145      /**
146       * This hook is called when a Sync Pipe shuts down, or when the Resync process
147       * shuts down. Any clean-up should be performed here.
148       * @param ctx
149       *          a TransactionContext which provides a valid JDBC connection to the
150       *          database.
151       */
152      @ThreadSafety(level = ThreadSafetyLevel.METHOD_NOT_THREADSAFE)
153      public void finalizeJDBCSyncDestination(
154                                          final TransactionContext ctx)
155      {
156        //no implementation required
157      }
158    
159      /**
160       * Return a full destination entry (in LDAP form) from the database,
161       * corresponding to the the source {@link Entry} that is passed in. This
162       * method should perform any queries necessary to gather the latest values for
163       * all the attributes to be synchronized and return them in an Entry.
164       * <p>
165       * Note that the if the source entry was renamed (see
166       * {@link SyncOperation#isModifyDN}), the <code>destEntryMappedFromSrc</code>
167       * will have the new DN; the old DN can be obtained by calling
168       * {@link SyncOperation#getDestinationEntryBeforeChange()} and getting the DN
169       * from there. This method should return the entry in its existing form
170       * (i.e. with the old DN, before it is changed).
171       * <p>
172       * This method <b>must be thread safe</b>, as it will be called repeatedly and
173       * concurrently by each of the Sync Pipe worker threads as they process
174       * entries.
175       * @param ctx
176       *          a TransactionContext which provides a valid JDBC connection to the
177       *          database.
178       * @param destEntryMappedFromSrc
179       *          the LDAP entry which corresponds to the database
180       *          "entry" to fetch
181       * @param  operation
182       *          the sync operation for this change
183       * @return a full LDAP Entry, or null if no such entry exists.
184       * @throws SQLException
185       *           if there is an error fetching the entry
186       */
187      @ThreadSafety(level = ThreadSafetyLevel.METHOD_THREADSAFE)
188      public abstract Entry fetchEntry(final TransactionContext ctx,
189                                       final Entry destEntryMappedFromSrc,
190                                       final SyncOperation operation)
191                                                  throws SQLException;
192    
193      /**
194       * Creates a full database "entry", corresponding to the the LDAP
195       * {@link Entry} that is passed in. This method should perform any inserts and
196       * updates necessary to make sure the entry is fully created on the database.
197       * <p>
198       * This method <b>must be thread safe</b>, as it will be called repeatedly and
199       * concurrently by the Sync Pipe worker threads as they process CREATE
200       * operations.
201       * @param ctx
202       *          a TransactionContext which provides a valid JDBC connection to the
203       *          database.
204       * @param entryToCreate
205       *          the LDAP entry which corresponds to the database "entry" to create
206       * @param  operation
207       *          the sync operation for this change
208       * @throws SQLException
209       *           if there is an error creating the entry
210       */
211      @ThreadSafety(level = ThreadSafetyLevel.METHOD_THREADSAFE)
212      public abstract void createEntry(final TransactionContext ctx,
213                                       final Entry entryToCreate,
214                                       final SyncOperation operation)
215                                          throws SQLException;
216    
217      /**
218       * Modify an "entry" in the database, corresponding to the the LDAP
219       * {@link Entry} that is passed in. This method may perform multiple updates
220       * (including inserting or deleting rows) in order to fully synchronize the
221       * entire entry on the database.
222       * <p>
223       * Note that the if the source entry was renamed (see
224       * {@link SyncOperation#isModifyDN}), the
225       * <code>fetchedDestEntry</code> will have the old DN; the new DN can
226       * be obtained by calling
227       * {@link SyncOperation#getDestinationEntryAfterChange()} and getting the DN
228       * from there.
229       * <p>
230       * This method <b>must be thread safe</b>, as it will be called repeatedly and
231       * concurrently by the Sync Pipe worker threads as they process MODIFY
232       * operations.
233       * @param ctx
234       *          a TransactionContext which provides a valid JDBC connection to the
235       *          database.
236       * @param fetchedDestEntry
237       *          the LDAP entry which corresponds to the database "entry" to modify
238       * @param modsToApply
239       *          a list of Modification objects which should be applied
240       * @param  operation
241       *          the sync operation for this change
242       * @throws SQLException
243       *           if there is an error modifying the entry
244       */
245      @ThreadSafety(level = ThreadSafetyLevel.METHOD_THREADSAFE)
246      public abstract void modifyEntry(final TransactionContext ctx,
247                                       final Entry fetchedDestEntry,
248                                       final List<Modification> modsToApply,
249                                       final SyncOperation operation)
250                                              throws SQLException;
251    
252      /**
253       * Delete a full "entry" from the database, corresponding to the the LDAP
254       * {@link Entry} that is passed in. This method may perform multiple deletes
255       * or updates if necessary to fully delete the entry from the database.
256       * <p>
257       * This method <b>must be thread safe</b>, as it will be called repeatedly and
258       * concurrently by the Sync Pipe worker threads as they process DELETE
259       * operations.
260       * @param ctx
261       *          a TransactionContext which provides a valid JDBC connection to the
262       *          database.
263       * @param fetchedDestEntry
264       *          the LDAP entry which corresponds to the database "entry" to delete
265       * @param  operation
266       *          the sync operation for this change
267       * @throws SQLException
268       *           if there is an error deleting the entry
269       */
270      @ThreadSafety(level = ThreadSafetyLevel.METHOD_THREADSAFE)
271      public abstract void deleteEntry(final TransactionContext ctx,
272                                       final Entry fetchedDestEntry,
273                                       final SyncOperation operation)
274                                            throws SQLException;
275    }
